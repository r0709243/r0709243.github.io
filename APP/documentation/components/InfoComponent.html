import { Component, OnInit, OnDestroy, Input } from '@angular/core';
import { VatService } from 'src/app/_services/vat.service';
import { DashboardService } from '../dashboard.service';
import { Subscription } from 'rxjs';

/**
 * Info component
 */
@Component({
  selector: 'app-info',
  templateUrl: './info.component.html',
  styleUrls: ['./info.component.scss']
})
export class InfoComponent implements OnInit, OnDestroy {

  /**
   * @ignore
   */
  private subscriptions: Subscription = new Subscription();

  /**
   * @ignore
   */
  @Input() vat: any;

  constructor(private _vatService: VatService, private _dashboardService: DashboardService) { }

  /**
   * Executes when the component is being initizalized
   */
  ngOnInit() {
    this.subscriptions.add(this._dashboardService.vatID.subscribe(result => {
      if (result != null) {
        this.subscriptions.add(this._vatService.getVatWithInfo(result).subscribe(result => {
          this.vat = result;
        }));
      }
    }));
  }

  /**
   * Executes when the component is being destroyed
   */
  ngOnDestroy() {
    this.subscriptions.unsubscribe();
  }


}
