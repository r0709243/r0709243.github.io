import { Component, OnInit, Input, OnDestroy } from '@angular/core';
import { DashboardService } from '../dashboard.service';
import { Subscription } from 'rxjs';
import { ManueleMetingService } from 'src/app/_services/manuele-meting.service';

/**
 * Measurements component
 */
@Component({
  selector: 'app-measurements',
  templateUrl: './measurements.component.html',
  styleUrls: ['./measurements.component.scss']
})
export class MeasurementsComponent implements OnInit, OnDestroy {

  /**
   * @ignore
   */
  @Input() measurements: any[]

  /**
   * @ignore
   */
  pageOfItems: Array<any>

  /**
   * @ignore
   */
  private subscriptions: Subscription = new Subscription()

  constructor(private _dashboardService: DashboardService, private _manueleMetingService: ManueleMetingService) { 
    this.subscriptions.add(this._dashboardService.refreshMeasurements.subscribe(result => {
      this.ngOnInit()
    }))
  }

  /**
   * Executes when the component is being initizalized
   */
  ngOnInit() {
    this.subscriptions.add(this._dashboardService.vatID.subscribe(result => {
      this.subscriptions.add(this._manueleMetingService.getManueleMetingenByVatID(result).subscribe(result => {
        this.measurements = result
      }))
    }))
  }

  /**
   * Executes when the component is being destroyed
   */
  ngOnDestroy() {
    this.subscriptions.unsubscribe()
  }

   /**
   * Executes when the page is changes
   * 
   * @param {Array<any>} pageOfItems page containing items
   */
  onChangePage(pageOfItems: Array<any>) {
    this.pageOfItems = pageOfItems
  }

}
