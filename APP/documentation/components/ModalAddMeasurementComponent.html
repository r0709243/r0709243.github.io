import { Component, OnInit, Input, OnDestroy } from '@angular/core';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import { NgbActiveModal } from '@ng-bootstrap/ng-bootstrap';
import { SoortMeting } from 'src/app/models/soort-meting.model';
import { ManueleMetingDTO } from 'src/app/models/DTOs/manuele-metingDTO.model';
import * as jwt_decode from 'jwt-decode';
import { ManueleMetingService } from 'src/app/_services/manuele-meting.service';
import { Subscription } from 'rxjs';
import { DatePipe } from '@angular/common';
import { DashboardService } from '../dashboard.service';

/**
 * Add measurement modal component
 */
@Component({
  selector: 'app-modal-add-measurement',
  templateUrl: './modal-add-measurement.component.html',
  styleUrls: ['./modal-add-measurement.component.scss'],
  providers: [DatePipe]
})
export class ModalAddMeasurementComponent implements OnInit, OnDestroy {

  /**
   * @ignore
   */
  @Input() units

  /**
   * @ignore
   */
  @Input() vatID

  /**
   * @ignore
   */
  private subscriptions: Subscription = new Subscription()

  /**
   * @ignore
   */
  addMeasurementForm: FormGroup

  /**
   * @ignore
   */
  submitted: boolean = false

  /**
   * @ignore
   */
  unitID: number

  /**
   * @ignore
   */
  dateNow = new Date()
  
  constructor(private formBuilder: FormBuilder, 
    public activeModal: NgbActiveModal, 
    private _manueleMetingService: ManueleMetingService, 
    private datePipe: DatePipe,
    private _dashboardService: DashboardService) {    
    this.addMeasurementForm = this.formBuilder.group({
      value: ['', Validators.required],
      date: [this.datePipe.transform(this.dateNow, 'yyyy-MM-dd')],
      time: [this.datePipe.transform(this.dateNow, 'HH:mm')]
    })
   }

   /**
   * Executes when the component is being initizalized
   */
  ngOnInit() {
    this.unitID = this.units[0].soortMetingID
  }

  /**
   * Executes when the component is being destroyed
   */
  ngOnDestroy() {
    this.subscriptions.unsubscribe()
  }

  /**
   * Executes when the form is submitted
   */
  onSubmit() {
    this.submitted = true

    if (this.addMeasurementForm.invalid) {
      return
    }

    let decoded = jwt_decode(localStorage.getItem("token"))
    let gebruikerID = decoded["GebruikerID"]

    let date = this.datePipe.transform(this.addMeasurementForm.get('date').value, 'yyyy-MM-dd')
    let time = this.addMeasurementForm.get('time').value

    let dateTime = date + " " + time

    let measurement = new ManueleMetingDTO(+gebruikerID, this.vatID, this.unitID, +this.addMeasurementForm.get('value').value, dateTime)

    console.log(dateTime)

    this.subscriptions.add(this._manueleMetingService.addManueleMeting(measurement).subscribe(result => {
      this.activeModal.close()
      this._dashboardService.measurementsRefresh("update")
    }))
  }

  /**
   * Executes when option changes from select tag
   * 
   * @param value input from user
   */
  onChange(value) {
    this.unitID = value
  }

  /**
   * Accessor for addMeasurementForm
   */
  get form() {
    return this.addMeasurementForm.controls;
  }

}
