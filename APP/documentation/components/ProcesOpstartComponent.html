  /**
   * Executes when the opstartform is submitted
   * Adds a new process the the database
   */
  onSubmitOpstart() {
    let naam = this.opstartForm.value.naam;
    let nummer = this.opstartForm.value.nummer;
    let druifID = this.opstartForm.value.druif.druifID;
    let oogst = this.opstartForm.value.oogst;
    let gekneusd = this.opstartForm.value.gekneusd;
    let locatie = this.opstartForm.value.locatie;
    let koelmantel = this.opstartForm.value.koelmantel;
    let mangat = this.opstartForm.value.mangat;
    let jaar = this.opstartForm.value.jaar;
    let typeVat = this.opstartForm.value.typeVat.typeVatID;
    let materiaal = this.opstartForm.value.materiaal.materiaalID;
    let soortwijn = this.opstartForm.value.soortwijn.soortWijnID;
    let programmaID = this.opstartForm.value.programma.programmaID;
    let piID = this.opstartForm.value.raspberryPi.raspberryPiID;

    this.pi = this.opstartForm.value.raspberryPi;
    this.pi.inGebruik = true;
    this.subscriptions.add(this._raspberryPiService.updateRaspberry(piID, this.pi).subscribe());

    let nieuwVat = new Vat(0, nummer, naam, oogst, gekneusd, programmaID, druifID, locatie, typeVat, koelmantel, materiaal, mangat, soortwijn, jaar, true, piID);
    this.subscriptions.add(this._vatService.nieuwProces(nieuwVat).subscribe(
      result => {
        this.nieuwVatID = result.vatID;

        if(this.opstartForm.value.alarmen){
          for(let item of this.opstartForm.value.alarmen){
            let onderdrempel = item.onderdrempel;
            let bovendrempel = item.bovendrempel;
            let sensorID = item.sensor;
            let alarmWaardeOnder = new AlarmWaarde(0, sensorID, onderdrempel, false)
            let alarmWaardeBoven = new AlarmWaarde(0, sensorID, bovendrempel, true)
            let alarmWaardeID: number;
            let vatID = this.nieuwVatID;

            this.subscriptions.add(this._alarmWaardeService.addAlarmWaarde(alarmWaardeOnder).subscribe(
              result => {
                alarmWaardeID = result.alarmWaardeID;
                for (let soortAlarmID of item.soortAlarm)
                  for (let userID of item.gebruiker){
                    let alarmGebruiker = new AlarmGebruiker(0, alarmWaardeID, soortAlarmID, vatID, userID);
                    this.subscriptions.add(this._alarmGebruikerService.addAlarmGebruiker(alarmGebruiker).subscribe());
                  }
              }
            ));
            this.subscriptions.add(this._alarmWaardeService.addAlarmWaarde(alarmWaardeBoven).subscribe(
              result => {
                alarmWaardeID = result.alarmWaardeID;
                for (let soortAlarmID of item.soortAlarm)
                  for (let userID of item.gebruiker){
                    let alarmGebruiker = new AlarmGebruiker(0, alarmWaardeID, soortAlarmID, vatID, userID);
                    this.subscriptions.add(this._alarmGebruikerService.addAlarmGebruiker(alarmGebruiker).subscribe());
                  }
              }
            ));
         }
        }
        
        if(this.opstartForm.value.acties){
          for(let item of this.opstartForm.value.acties){
            let naam = item.actie;
            let datum = item.datum;
            let tijd = item.tijd;
            let soortActie = item.soortActie;
            let notitie = item.notitie;
            let datetime = datum + " " + tijd;
            let gebruikerID = parseInt(localStorage.getItem("gebruikerID"));
            let actie = new Actie(0, naam, datetime, this.nieuwVatID, gebruikerID, soortActie, notitie);
            this.subscriptions.add(this.subscriptions.add(this._actieService.addActie(actie).subscribe()));
          }
        }
        
        if(this.opstartForm.value.automatischeMetingen){
          let sensorIDs: number[] = [];

          for (let item of this.opstartForm.value.automatischeMetingen){
            sensorIDs.push(item);
          }

          for (let sensor of this.sensors){
            if (sensorIDs.includes(sensor.sensorID)){
              sensor.isActief = true;
            } else {
              sensor.isActief = false;
            }
            this.subscriptions.add(this._sensorService.updateSensor(sensor.sensorID, sensor).subscribe());
          }  
        }
        this.router.navigate(['dashboard', this.nieuwVatID]);
      }  
    ));
    alert("Nieuw proces gestart.");
  }
