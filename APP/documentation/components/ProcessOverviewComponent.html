import { Component, OnInit, Output, OnDestroy } from '@angular/core';
import { ActivatedRoute } from '@angular/router';
import { Subscription } from 'rxjs';
import { filter, startWith } from 'rxjs/operators';
import { CardStyle1Component } from './cards-templates/card-style1/card-style1.component';
import { CardStyle2Component } from './cards-templates/card-style2/card-style2.component';

/**
 * Process overview component
 */
@Component({
  selector: 'app-process-overview',
  templateUrl: './process-overview.component.html',
  styleUrls: ['./process-overview.component.scss']
})
export class ProcessOverviewComponent implements OnInit, OnDestroy {

  /**
   * @ignore
   */
  private subscriptions: Subscription = new Subscription()

  /**
   * @ignore
   */
  tanks: any[]

  /**
   * @ignore
   */
  filteredTanks: any[]

  /**
   * @ignore
   */
  name: string = ''

  constructor(private activatedRoute: ActivatedRoute) { }

  /**
   * Executes when the component is being initizalized
   */
  ngOnInit() {
    this.subscriptions.add(this.activatedRoute.data.subscribe((result: { tanks: any[] }) => {
      this.tanks = result.tanks

      result.tanks.forEach(x => {

        if (x.isActief == true) {
          x.template = CardStyle1Component
        } else {
          x.template = CardStyle2Component
        }
      })
    }))

    this.filterActive()
  }

  /**
   * Executes when the component is being destroyed
   */
  ngOnDestroy() {
    this.subscriptions.unsubscribe()
  }

  /**
   * Executes when the filter is changed
   * 
   * @param filterType variable containing how to filter the tank array
   */
  onFilter(filterType: string) {
    if (filterType == 'all') {
      this.filterAll()
    }
    if (filterType == 'active') {
      this.filterActive()
    }
    if (filterType == 'inactive') {
      this.filterInactive()
    }
  }

  /**
   * returns all the tanks
   */
  filterAll() {
    this.filteredTanks = this.tanks
  }

  /**
   * filters the array tanks if it is active
   */
  filterActive() {
    this.filteredTanks = this.tanks.filter(x => x.isActief == true)
  }

  /**
   * filters the array tanks if it is inactive
   */
  filterInactive() {
    this.filteredTanks = this.tanks.filter(x => x.isActief == false)
  }


}
