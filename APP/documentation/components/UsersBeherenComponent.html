import { Component, OnInit } from '@angular/core';
import { FormBuilder, FormGroup, AbstractControl, Validators } from '@angular/forms';
import { GebruikerService } from '../_services/gebruiker.service';
import { Gebruiker } from '../models/gebruiker.model';
import { Subscription } from 'rxjs';
import { NgbModal } from '@ng-bootstrap/ng-bootstrap';

/**
 * Users-beheren component
 */
@Component({
  selector: 'app-users-beheren',
  templateUrl: './users-beheren.component.html',
  styleUrls: ['./users-beheren.component.scss']
})
export class UsersBeherenComponent implements OnInit {

  /**
   * @ignore
   */
  private subscriptions: Subscription = new Subscription();
  /**
   * @ignore
   */
  gebruikers: Gebruiker[];
  /**
   * @ignore
   */
  gebruikerID: any;
  /**
   * @ignore
   */
  currentGebruiker: Gebruiker;

  /**
   * @ignore
   */
  userForm: FormGroup;
  /**
   * @ignore
   */
  delForm: FormGroup;

  /**
   * @ignore
   */
  addUserForm = this.fb.group({
    firstname: ['', Validators.required],
    lastname: ['', Validators.required],
    email: ['', [Validators.required, Validators.email]],
    isAdmin: [false],
    phone: ['', [Validators.required, Validators.pattern("^[0-9]*$")]],
    passw: ['', Validators.required],
    passwval: ['', [Validators.required]]
  }, {validator: this.matchPasswordAdd});


  constructor(
    private fb:FormBuilder,
    private _gebruikerService: GebruikerService,
    private modalService: NgbModal
  ) { }

  /**
   * Executes when the component is initialized
   */
  ngOnInit() {
    this.getAllUsers();
  }

  /**
   * Gets all users from the database
   */
  getAllUsers(){
    this.gebruikerID = localStorage.getItem("gebruikerID");
    this.subscriptions.add(this._gebruikerService.getAllGebruikers().subscribe(
      result => {
        this.gebruikers = result;
        this.currentGebruiker = this.gebruikers.find(x => x.gebruikerID == this.gebruikerID);
        this.gebruikers.splice(this.gebruikers.findIndex(x => x.gebruikerID == this.gebruikerID), 1);
      }
    ));
  }

  /**
   * Executes when the component is destroyed
   */
  ngOnDestroy() {
    this.subscriptions.unsubscribe();
  }

  /**
   * @ignore
   */
  editUser(user: Gebruiker, content){
    this.modalService.open(content, { ariaLabelledBy: 'modal-user-title' });

    this.userForm = this.fb.group({
      naam: [user.naam, Validators.required],
      email: [user.email, [Validators.required, Validators.email]],
      telNummer: [user.telNummer, [Validators.required, Validators.pattern("^[0-9]*$")]],
      isAdmin: [user.isAdmin],
      passwChange: [false],
      passwNew: [''],
      passwNewConfirm: [''],
      user: [user]
    }, {validator: this.matchPasswordEdit})
  }

  /**
   * Executes when the edit user form is submitted
   * Edits a user record in the database
   */
  onSubmitEditUser(){
    let user = this.userForm.value.user;
    user.naam = this.userForm.value.naam;
    user.email = this.userForm.value.email;
    user.telNummer = this.userForm.value.telNummer;
    user.isAdmin = this.userForm.value.isAdmin;
    if (this.userForm.value.passwChange == true){
      user.wachtwoord = this.userForm.value.passwNew;
      this.subscriptions.add(this._gebruikerService.putGebruiker(user.gebruikerID, user).subscribe());
    } else {
      this.subscriptions.add(this._gebruikerService.putGebruikerNoPassw(user.gebruikerID, user).subscribe());
    }
  }

  /**
   * Checks if the given passwords match when creating a new user
   */
  matchPasswordAdd(control: AbstractControl){
    let passw = control.get('passw').value;
    let passwval = control.get('passwval').value;
    if (passw != passwval){
      control.get('passwval').setErrors({MatchPassword: true});
    }else{
      return null;
    }
  }

  /**
   * Checks if the given passwords match when editing a user
   */
  matchPasswordEdit(control: AbstractControl){
    let passw = control.get('passwNew').value;
    let passwval = control.get('passwNewConfirm').value;
    if (passw != passwval){
      control.get('passwNewConfirm').setErrors({MatchPassword: true});
    }else{
      return null;
    }
  }

  /**
   * Executes when the password checkbox checkstate changes
   * Sets validators for password depending on the checkstate
   */
  passwCheckbox(){
    if (!this.userForm.value.passwChange){
      this.userForm.get('passwNew').setValidators(Validators.required);
      this.userForm.get('passwNewConfirm').setValidators(Validators.required);
    } else {
      this.userForm.get('passwNew').clearValidators();
      this.userForm.get('passwNew').updateValueAndValidity();
      this.userForm.get('passwNewConfirm').clearValidators();
      this.userForm.get('passwNewConfirm').updateValueAndValidity();
    }
  }

  /**
   * Opens the confirmation form for deleteing a user
   */
  deleteUser(user: Gebruiker, content){
    this.modalService.open(content, { ariaLabelledBy: 'modal-delUser-title' });

    this.delForm = this.fb.group({
      naam: [user.naam],
      email: [user.email],
      telNummer: [user.telNummer],
      isAdmin: [user.isAdmin],
      user: [user]
    })
  }

  /**
   * Executes when the delete user form is submitted
   * Deletes a user from the database
   */
  onSubmitDelUser(){
    let user = this.delForm.value.user;
    this.subscriptions.add(this._gebruikerService.deleteGebruiker(user.gebruikerID).subscribe(
      result => {
        this.getAllUsers();
      }
    ));
  }

  /**
   * @ignore
   */
  onModalUserClose(){
    this.userForm.reset();
  }

  /**
   * @ignore
   */
  onModalDelUserClose(){
    this.delForm.reset();
  }

  /**
   * @ignore
   */
  onModalAddUserClose(){
    this.addUserForm.reset();
  }

  /**
   * @ignore
   */
  openAddUserModal(content){
    this.modalService.open(content, { ariaLabelledBy: 'modal-addUser-title' });
  }

  /**
   * Executes when the form for creating a user is submitted
   * Creates a new user in the database
   */
  onSubmitAddUser(){
    let naam = this.addUserForm.value.firstname + ' ' + this.addUserForm.value.lastname;
    let email = this.addUserForm.value.email;
    let telNummer = this.addUserForm.value.phone;
    let isAdmin = this.addUserForm.value.isAdmin;
    let wachtwoord = this.addUserForm.value.passw;
    let newUser = new Gebruiker(0,naam, wachtwoord, email, telNummer, isAdmin, "");
    this.subscriptions.add(this._gebruikerService.addGebruiker(newUser).subscribe(
      result => {
        this.getAllUsers();
      }
    ));
    
  }
}
