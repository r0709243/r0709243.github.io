import { Component, OnInit, OnDestroy, Input } from '@angular/core';
import { NgbActiveModal } from '@ng-bootstrap/ng-bootstrap';
import { FormBuilder, FormGroup, Validators } from '@angular/forms';
import * as jwt_decode from 'jwt-decode';
import { ActieDTO } from 'src/app/models/DTOs/actie-dto.service';
import { ActieService } from 'src/app/_services/actie.service';
import { Subscription } from 'rxjs';
import { DatePipe } from '@angular/common';
import { Actie } from 'src/app/models/actie.model';
import { DashboardService } from '../dashboard.service';
import { dashCaseToCamelCase } from '@angular/compiler/src/util';

/**
 * Add activity modam component
 */
@Component({
  selector: 'app-modal-add-activity',
  templateUrl: './modal-add-activity.component.html',
  styleUrls: ['./modal-add-activity.component.scss'],
  providers: [DatePipe]
})
export class ModalAddActivityComponent implements OnInit, OnDestroy {

  /**
   * @ignore
   */
  @Input() actions

  /**
   * @ignore
   */
  @Input() vatID

  /**
   * @ignore
   */
  private subscriptions: Subscription = new Subscription()

  /**
   * @ignore
   */
  addActionForm: FormGroup

  /**
   * @ignore
   */
  actionID: number

  /**
   * @ignore
   */
  submitted: boolean = false

  /**
   * @ignore
   */
  dateNow = new Date()

  constructor(public activeModal: NgbActiveModal, 
    private formBuilder: FormBuilder, 
    private _actieService: ActieService, 
    private datePipe: DatePipe,
    private _dashboardService: DashboardService) {
    this.addActionForm = this.formBuilder.group({
      name: ['', Validators.required],
      date: [this.datePipe.transform(this.dateNow, 'yyyy-MM-dd')],
      time: [this.datePipe.transform(this.dateNow, 'HH:mm')],
      note: ['']
    })
   }

   /**
   * Executes when the component is being initizalized
   */
  ngOnInit() {
    this.actionID = this.actions[0].soortActieID
  }

  /**
   * Executes when the component is being destroyed
   */
  ngOnDestroy() {
    this.subscriptions.unsubscribe()
  }

  /**
   * Executes when option changes from select tag
   * 
   * @param value input from user
   */
  onChange(value) {
    this.actionID = value
  }

  /**
   * Accessor for addActionForm
   */
  get form() {
    return this.addActionForm.controls;
  }

  /**
   * Executes when the form is submitted
   */
  onSubmit() {
    this.submitted = true

    if (this.addActionForm.invalid) {
      return
    }

    let decoded = jwt_decode(localStorage.getItem("token"))
    let gebruikerID = decoded["GebruikerID"]

    let date = this.datePipe.transform(this.addActionForm.get('date').value, 'yyyy-MM-dd')
    let time = this.addActionForm.get('time').value

    let dateTime = date + " " + time

    let action = new Actie(0,this.addActionForm.get('name').value, dateTime, this.vatID, +gebruikerID, this.actionID, this.addActionForm.get('note').value)
    
    this.subscriptions.add(this._actieService.addActie(action).subscribe(result => {
      this.activeModal.close()
      this._dashboardService.actionsRefresh("update")
    }))
  }

}
