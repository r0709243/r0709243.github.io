import { Component, OnInit, OnDestroy, Input } from '@angular/core';
import { NgbActiveModal } from '@ng-bootstrap/ng-bootstrap';
import { VatService } from 'src/app/_services/vat.service';
import { Subscription } from 'rxjs';
import { DashboardService } from '../dashboard.service';
import { RaspberryPiService } from 'src/app/_services/raspberry-pi.service';

/**
 * Stop process modal component
 */
@Component({
  selector: 'app-modal-stop-process',
  templateUrl: './modal-stop-process.component.html',
  styleUrls: ['./modal-stop-process.component.scss']
})
export class ModalStopProcessComponent implements OnInit, OnDestroy {

  /**
   * @ignore
   */
  @Input() vatID

  /**
   * @ignore
   */
  @Input() piID

  /**
   * @ignore
   */
  private subscriptions: Subscription = new Subscription()

  constructor(
    public activeModal: NgbActiveModal,
    private _vatService: VatService,
    private _dashboardService: DashboardService,
    private _raspberryPiService: RaspberryPiService) { }

  /**
   * Executes when the component is being initizalized
   */
  ngOnInit() {
  }

  /**
   * Executes when the component is being destroyed
   */
  ngOnDestroy() {
    this.subscriptions.unsubscribe()
  }

  /**
   * stops the winemaking process
   */
  stopProcess() {
    this.subscriptions.add(this._vatService.deactivateVat(this.vatID).subscribe(result => {
      this.activeModal.close()
      this._dashboardService.refresh("update")
    }))


  }

}

<div class="modal-header">
