import { Component, OnInit, OnDestroy, Input } from '@angular/core';
import { VatService } from 'src/app/_services/vat.service';
import { Vat } from 'src/app/models/vat.model';
import { Subscription } from 'rxjs';
import { DashboardService } from '../dashboard.service';
import { NgbModal } from '@ng-bootstrap/ng-bootstrap';
import { ModalStopProcessComponent } from '../modal-stop-process/modal-stop-process.component'
import { ModalAddMeasurementComponent } from '../modal-add-measurement/modal-add-measurement.component'
import { SoortMetingService } from 'src/app/_services/soort-meting.service';
import { ActieService } from 'src/app/_services/actie.service';
import { ModalAddActivityComponent } from '../modal-add-activity/modal-add-activity.component';
import * as jwt_decode from 'jwt-decode';

/**
 * Navigation component
 */
@Component({
  selector: 'app-navigation',
  templateUrl: './navigation.component.html',
  styleUrls: ['./navigation.component.scss']
})
export class NavigationComponent implements OnInit, OnDestroy {

  /**
   * @ignore
   */
  private subscriptions: Subscription = new Subscription()

  /**
   * @ignore
   */
  @Input() vaten: Vat[]

  /**
   * @ignore
   */
  vatID: number

  /**
   * @ignore
   */
  isAdmin: string = ''

  constructor(private _vatService: VatService, private _dashboardService: DashboardService, private _modalService: NgbModal, private _soortMetingService: SoortMetingService
    , private _actieService: ActieService) { 
    this.subscriptions.add(this._dashboardService.refreshNavigation.subscribe(result => {
      this.ngOnInit()
    }))
  }

  /**
   * Executes when the component is being initizalized
   */
  ngOnInit() {
    this.subscriptions.add(this._vatService.getAllActiveVaten().subscribe(result => {
      this.vaten = result
      this._dashboardService.updateVatID(result[0].vatID)
      this.vatID = result[0].vatID
    }));

    let token = localStorage.getItem("token")
    let decoded = jwt_decode(token)
    if(decoded['IsAdmin'] == 'True') {
      this.isAdmin = 'isAdmin'
    }
  }

  /**
   * Executes when the component is being destroyed
   */
  ngOnDestroy() {
    this.subscriptions.unsubscribe();
  }

  /**
   * Executes when option changes from select tag
   * 
   * @param vatID input from user
   */
  onChange(vatID) {
    this._dashboardService.updateVatID(vatID)
    this.vatID = vatID
  }

  /**
   * Shows the stop-process modal
   */
  showStopProcessModal() {
    const modalRef = this._modalService.open(ModalStopProcessComponent)
    modalRef.componentInstance.vatID = this.vatID
  }

  /**
   * Shows the add-measurement modal
   */
  showAddMeasurementModal() {
    this.subscriptions.add(this._soortMetingService.getAllSoortMetingen().subscribe(result => {
      const modalRef = this._modalService.open(ModalAddMeasurementComponent)
      modalRef.componentInstance.units = result
      modalRef.componentInstance.vatID = this.vatID
    }))
  }

  /**
   * Shows the add-action modal
   */
  showAddActionModal() {
    this.subscriptions.add(this._actieService.getAllSoortenActies().subscribe(result => {
      const modalRef = this._modalService.open(ModalAddActivityComponent)
      modalRef.componentInstance.actions = result
      modalRef.componentInstance.vatID = this.vatID
    }))
  }

}
