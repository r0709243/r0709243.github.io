import { Component, OnInit, OnDestroy } from '@angular/core';
import { Title } from '@angular/platform-browser';
import { Router } from '@angular/router';
import { LoginService } from './login/login.service';
import { Subscription } from 'rxjs';
import { VatService } from './_services/vat.service';

/**
 * App component
 */
@Component({
  selector: 'app-root',
  templateUrl: './app.component.html',
  styleUrls: ['./app.component.scss']
})
export class AppComponent implements OnInit, OnDestroy{

  /**
   * @ignore
   */
  loggedIn: boolean = false
  /**
   * @ignore
   */
  loggedOut: boolean = true

  /**
   * @ignore
   */
  private subscription: Subscription = new Subscription()

  constructor(private _titleService: Title,
    private _router: Router,
    private _loginService: LoginService,
    private _vatService: VatService) {
  }

  /**
   * Executes when the component is initialized
   */
  ngOnInit() {
    this._titleService.setTitle('Rewined');

    if (this._loginService.loggedIn) {
      this.loggedIn = true
      this.loggedOut = false
      this.subscription.add(this._vatService.getAllVaten().subscribe(result => {
        this._router.navigate(['dashboard', result[0].vatID])
      }))
    }

    this.subscription.add(this._loginService.loggedIn.subscribe(result => {
      if (result) {
        this.loggedIn = true;
        this.loggedOut = false;
      }
    }))
  }

  /**
   * Executes when the component is destroyed
   */
  ngOnDestroy() {
    this.subscription.unsubscribe()
  }

  /**
   * Logs the user out
   */
  logout() {
    this._loginService.isLoggingOut()
    this.loggedIn = false
    this.loggedOut = true
    this._router.navigate([''])
  }

  /**
   * Navigates to the dashboard
   */
  dashboard() {
    this.subscription.add(this._vatService.getAllVaten().subscribe(result => {
      this._router.navigate(['dashboard', result[0].vatID])
    }))
  }

}
