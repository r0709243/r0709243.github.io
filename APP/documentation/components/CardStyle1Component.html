import { Component, OnInit, Pipe, PipeTransform } from '@angular/core';
import { CardTemplateBaseComponent } from '../cards-template-base';
import { DomSanitizer } from '@angular/platform-browser';
import { environment } from '../../../../environments/environment';


/**
 * Card style 1 component
 */
@Component({
  selector: 'app-card-style1',
  templateUrl: './card-style1.component.html',
  styleUrls: ['./card-style1.component.scss']
})
export class CardStyle1Component extends CardTemplateBaseComponent implements OnInit {

  /**
   * @ignore
   */
  temperatureUrl = environment.generalTemperatureBaseUrl

  /**
   * @ignore
   */
  pressureUrl = environment.pressureBaseUrl

  /**
   * @ignore
   */
  alcoholUrl = environment.generalAlcoholBaseUrl

  /**
   * @ignore
   */
  temperature: string

  /**
   * @ignore
   */
  pressure: string

  /**
   * @ignore
   */
  alcohol: string

  /**
   * Executes when the component is initialized
   */
  ngOnInit() {
    this.temperature = this.temperatureUrl + this.data.tank.vatID
    this.pressure = this.pressureUrl + this.data.tank.vatID
    this.alcohol = this.alcoholUrl + this.data.tank.vatID
  }
}
