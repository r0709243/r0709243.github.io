import { Component, OnInit, OnDestroy } from '@angular/core';
import { Validators, FormBuilder, AbstractControl } from '@angular/forms';
import { LoginService } from './login.service';
import { UserLogin } from '../models/user-login.model';
import { Title } from '@angular/platform-browser';
import { Router } from '@angular/router';
import { Subscription } from 'rxjs';
import { VatService } from '../_services/vat.service';

/**
 * Login component
 */
@Component({
  selector: 'app-login',
  templateUrl: './login.component.html',
  styleUrls: ['./login.component.scss']
})
export class LoginComponent implements OnInit, OnDestroy {

  /**
   * @ignore
   */
  private subscriptions: Subscription = new Subscription()

  /**
   * @ignore
   */
  inlogForm = this.fb.group({
    email: ['', [Validators.required, Validators.email]],
    passw: ['', Validators.required]
  })

  constructor(private fb: FormBuilder, private _loginService: LoginService, private _titleService: Title, private _router: Router,
     private _vatService: VatService) { }

  /**
   * Executes when the component is initialized
   */
  ngOnInit() {
    this._titleService.setTitle("Rewined - Login")
  }

  /**
   * Executes when the component is destroyed
   */
  ngOnDestroy() {
    this.subscriptions.unsubscribe()
  }

  /**
   * Executes when the login form is submitted
   * Logs a user in if the given credentials are found in the database
   */
  onSubmitLogin(){
    let email = this.inlogForm.value.email
    let wachtwoord = this.inlogForm.value.passw
    let loginUser = new UserLogin(email, wachtwoord)
    this.subscriptions.add(this._loginService.authenticate(loginUser).subscribe(
      data => {
        localStorage.setItem("gebruikerID", data.gebruikerID.toString());
        localStorage.setItem("token", data.token);
        this._loginService.updateLoggedIn(true);
        this.subscriptions.add(this._vatService.getAllVaten().subscribe(result => {
          this._router.navigate(['dashboard', result[0].vatID])
        }))
      },
      error => alert('Email or password are wrong!')
    ))
  }
}
