import { Component, OnInit, Input, OnDestroy } from '@angular/core';
import { DashboardService } from '../dashboard.service';
import { Subscription } from 'rxjs';
import { ActieService } from 'src/app/_services/actie.service';

/**
 * Actions component
 */
@Component({
  selector: 'app-actions',
  templateUrl: './actions.component.html',
  styleUrls: ['./actions.component.scss']
})
export class ActionsComponent implements OnInit, OnDestroy {

  /**
   * @ignore
   */
  @Input() actions

  /**
   * @ignore
   */
  pageOfItems: Array<any>

  /**
   * @ignore
   */
  private subscriptions: Subscription = new Subscription()

  constructor(
    private _dashboardService: DashboardService, 
    private _actieService: ActieService) { 
    this.subscriptions.add(this._dashboardService.refreshActions.subscribe(result => {
      this.ngOnInit()
    }))
  }

  /**
   * Executes when the component is being initizalized
   */
  ngOnInit() {
    this.subscriptions.add(this._dashboardService.vatID.subscribe(result => {
      this.subscriptions.add(this._actieService.getActiesByVatID(result).subscribe(result => {
        this.actions = result
      }))
    }))
  }

  /**
   * Executes when the component is being destroyed
   */
  ngOnDestroy() {
    this.subscriptions.unsubscribe()
  }

  /**
   * Executes when the page is changes
   * 
   * @param {Array<any>} pageOfItems page containing items
   */
  onChangePage(pageOfItems: Array<any>) {
    this.pageOfItems = pageOfItems
  }

}
