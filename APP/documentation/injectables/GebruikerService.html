import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Observable } from 'rxjs';
import { Gebruiker } from '../models/gebruiker.model';
import { environment } from '../../environments/environment';

/**
 * Gebruiker service
 * 
 * Includes all GET, POST, PUT and DELETE calls and BehaviorSubjects
 */
@Injectable({
  providedIn: 'root'
})
export class GebruikerService {

  /**
   * Base url for api calls
   */
  baseUrl = environment.baseUrl;

  constructor(private http: HttpClient) { }

  /**
   * Get all the users
   * 
   * @returns all gebruikers
   */
  getAllGebruikers(): Observable<Gebruiker[]>{
    return this.http.get<Gebruiker[]>(this.baseUrl + "Gebruiker");
  }

  /**
   * Update a user
   * 
   * @param {number} id id of the record that needs to be updated
   * @param {Gebruiker} user updated record
   */
  putGebruiker(id:number, user: Gebruiker){
    return this.http.put<Gebruiker>(this.baseUrl + "Gebruiker/" + id, user);
  }

  /**
   * Update a user without changed password
   * 
   * @param id id of the record that needs to be updated 
   * @param user updated record
   */
  putGebruikerNoPassw(id:number, user: Gebruiker){
    return this.http.put<Gebruiker>(this.baseUrl + "Gebruiker/putNoPass/" + id, user);
  }

  /**
   * Delete a user
   * 
   * @param id id of the record that needs to be deleted
   * @returns deleted record
   */
  deleteGebruiker(id:number){
    return this.http.delete<Gebruiker>(this.baseUrl + "Gebruiker/" + id);
  }

  /**
   * Add a new user
   * 
   * @param {Gebruiker} gebruiker new record
   * @returns id from added record
   */
  addGebruiker(gebruiker: Gebruiker){
    return this.http.post<Gebruiker>(this.baseUrl + "Gebruiker", gebruiker);
  }
}
