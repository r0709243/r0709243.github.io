import { Injectable } from '@angular/core';
import { environment } from '../../environments/environment';
import { ManueleMetingDTO } from '../models/DTOs/manuele-metingDTO.model';
import { Observable } from 'rxjs';
import { HttpClient } from '@angular/common/http';

/**
 * Manuele meting service
 * 
 * Includes all GET, POST, PUT and DELETE calls and BehaviorSubjects
 */
@Injectable({
  providedIn: 'root'
})
export class ManueleMetingService {

  /**
   * Base url for api calls
   */
  baseUrl = environment.baseUrl

  constructor(private http: HttpClient) { }

  /**
   * Add a new manuele meting
   * 
   * @param {ManueleMetingDTO} meting new record
   * @returns id from added record
   */
  addManueleMeting(meting: ManueleMetingDTO): Observable<any> {
    return this.http.post<any>(this.baseUrl + "ManueleMeting", meting)
  }

  /**
   * Get all the manuele metingen based of this vatID
   * 
   * @param {number} vatID id to find the record
   * @returns all manuele metingen
   */
  getManueleMetingenByVatID(vatID: number): Observable<any[]> {
    return this.http.get<any[]>(this.baseUrl + "ManueleMeting/metingen/vat/" + vatID)
  }
}
