import { Injectable } from '@angular/core';
import { HttpClient, HttpHeaders } from '@angular/common/http';
import { UserLogin } from '../models/user-login.model';
import { Observable, BehaviorSubject } from 'rxjs';
import { Gebruiker } from '../models/gebruiker.model';
import * as jwt_decode from 'jwt-decode';
import { environment } from 'src/environments/environment';

/**
 * Login service
 */
@Injectable({
  providedIn: 'root'
})
export class LoginService {

  /**
   * Base url for api calls
   */
  baseUrl = environment.baseUrl

  constructor(private http: HttpClient) { }

  /**
   * Check if the user credentials are valid
   * 
   * @param {UserLogin} userLogin user credentials
   * @returns a jwt token if valid
   */
  authenticate(userLogin: UserLogin): Observable<Gebruiker>{
    return this.http.post<Gebruiker>( this.baseUrl + "Gebruiker/authenticate", userLogin);
  }

  /**
   * Checks if a user is loggen in
   */
  isLoggedIn(){
    let token = localStorage.getItem("token");
    if(token){
      return true;
    } else{
      return false;
    }
  }

  /**
   * Empties the token when a user is logging out
   */
  isLoggingOut() {
    localStorage.clear();
  }

  /**
   * Check if a user is a admin
   */
  isAdmin(){
    let token = localStorage.getItem("token");
    if (token){
      let decoded = jwt_decode(token);
      if(decoded['IsAdmin'] == 'True'){
        return true;
      } 
    }
    return false;
  }

  /**
   * @ignore
   */
  private loggedInSubject = new BehaviorSubject(false);
  /**
   * @ignore
   */
  loggedIn = this.loggedInSubject.asObservable();
  /**
   * Update the status
   * 
   * @param status new status
   */
  updateLoggedIn(status: boolean) {
    this.loggedInSubject.next(status);
  }

}
