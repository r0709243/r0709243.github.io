import { Injectable } from '@angular/core';
import { Actie } from '../models/actie.model';
import { environment } from '../../environments/environment';
import { Observable } from 'rxjs';
import { HttpClient } from '@angular/common/http';
import { ActieDTO } from '../models/DTOs/actie-dto.service';
import { ObserveOnSubscriber } from 'rxjs/internal/operators/observeOn';

/**
 * Actie service
 * 
 * Includes all GET, POST, PUT and DELETE calls and BehaviorSubjects
 */
@Injectable({
  providedIn: 'root'
})
export class ActieService {

  /**
   * Base url for api calls
   */
  baseUrl = environment.baseUrl;

  constructor(private http: HttpClient) { }

  /**
   * Get all the acties
   * 
   * @returns all acties
   */
  getActies(): Observable<Actie[]>{
    return this.http.get<Actie[]>(this.baseUrl + "actie");
  }

  /**
   * Get all soort-acties
   * 
   * @returns all soort-acties
   */
  getAllSoortenActies(): Observable<any[]> {
    return this.http.get<any[]>(this.baseUrl + "SoortActie")
  }

  /**
   * Add a new actie
   * 
   * @param {Actie} actie new record
   * @returns id from added record
   */
  addActie(actie: Actie): Observable<any> {
    return this.http.post<any>(this.baseUrl + "Actie", actie)
  }

  /**
   * Get all acties from a tank
   * 
   * @param {number} vatID variable of the tank id
   * @returns all the actions/events from a tank
   */
  getActiesByVatID(vatID: number): Observable<any[]> {
    return this.http.get<any[]>(this.baseUrl + "Actie/acties/vat/" + vatID)
  }
}
