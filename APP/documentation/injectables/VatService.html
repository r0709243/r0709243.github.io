import { Injectable } from '@angular/core';
import { HttpClient } from '@angular/common/http';
import { Vat } from '../models/vat.model';
import { Observable } from 'rxjs';
import { environment } from '../../environments/environment';

/**
 * Vat service
 * 
 * Includes all GET, POST, PUT and DELETE calls and BehaviorSubjects
 */
@Injectable({
  providedIn: 'root'
})
export class VatService {

  /**
   * Base url for api calls
   */
  baseUrl = environment.baseUrl;

  constructor(private http: HttpClient) { }
 
  /**
   * Get all vaten
   * 
   * @returns all vaten
   */
  getAllVaten(): Observable<Vat[]>{
    return this.http.get<Vat[]>(this.baseUrl + "Vat")
  }

  /**
   * Get all active vaten
   * 
   * @returns all active vaten
   */
  getAllActiveVaten(): Observable<Vat[]> {
    return this.http.get<Vat[]>(this.baseUrl + "Vat/active")
  }

  /**
   * Add new process
   * 
   * @param {Vat} vat new record
   * @returns id from added record
   */
  nieuwProces(vat: Vat) {
    return this.http.post<Vat>(this.baseUrl + "Vat", vat)
  }

  /**
   * Get vat with additional info by id
   * 
   * @param {number} id record with this id
   * @returns vat with additional information
   */
  getVatWithInfo(id: number) : Observable<any> {
    return this.http.get<any>(this.baseUrl + "Vat/info/" + id)
  }

  /**
   * Update a vat to inactive
   * 
   * @param {number} id id of that record that needs to be updated
   */
  deactivateVat(id: number) {
    return this.http.put<any>(this.baseUrl + "Vat/deactivate/" + id, null)
  }

  /**
   * Get all vaten with additional info
   * 
   * @returns all vaten with additional information
   */
  getAllVatenWithInfo(): Observable<any[]> {
    return this.http.get<any[]>(this.baseUrl + "Vat/info/all")
  }
}
